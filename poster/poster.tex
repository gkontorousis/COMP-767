\begin{alertblock}{Key Message}
\justifying
We formalize Occam's razor for linguistic explanations: prefer prompts that are simultaneously \textbf{concise} and \textbf{predictive}. Minimum Description Length (MDL) provides this principle rigorously, and we train an amortized RL policy to discover such explanations across datasets.
\end{alertblock}

\begin{block}{Motivation}
\justifying
Prompted LLMs act as universal function approximators~\cite{petrov2024} --- prompts play the role of natural-language ``parameters'' determining model behavior.

\begin{itemize}
  \item Current approaches are \textbf{heuristic}: manual prompt engineering, post-hoc requests, or unconstrained generation
  \item Long explanations may overfit; short ones may underfit --- no principled trade-off exists
  \item We formalize Occam's razor: prefer explanations that are (i) concise \emph{and} (ii) predictive
\end{itemize}
\end{block}

\begin{block}{MDL Framework}
\justifying
An explanation $z$ is a natural-language model of dataset $x$. A frozen evaluator LLM $p^*$ scores $z$ using a two-part code:

\begin{equation}
\mathrm{MDL}(z, x) = \underbrace{-\log p^*(z)}_{\text{explanation complexity}} + \underbrace{-\log p^*(x \mid z)}_{\text{data complexity given } z}
\end{equation}

\textbf{Bayesian interpretation:} minimizing MDL $\equiv$ maximizing the posterior $p^*(z \mid x)$.

\begin{itemize}
  \item $-\log p^*(z)$: penalizes verbose or unlikely explanations --- \emph{simplicity pressure}
  \item $-\log p^*(x \mid z)$: penalizes explanations that fail to predict data --- \emph{accuracy pressure}
  \item Mask $m$ restricts likelihood to prediction-relevant tokens only (e.g.\ labels, next-state tokens)
\end{itemize}
\end{block}

\begin{block}{Method Overview}
\justifying
Dataset $x$ $\to$ Policy $p_\theta(z|x)$ $\to$ Evaluator $p^*$ $\to$ Reward $R$ $\xrightarrow{\text{RL}}$ update $p_\theta$

\begin{itemize}
  \item \textbf{Policy} $p_\theta$: fine-tuned LLM; the only model updated during training
  \item \textbf{Evaluator} $p^*$: frozen; computes prior NLL $-\!\log p^*(z)$ and likelihood NLL $-\!\log p^*(x|z)$
  \item \textbf{Amortized}: once trained, $p_\theta$ generates explanations for new datasets in one forward pass
\end{itemize}
\end{block}

\begin{block}{Reward Signal \& RL Training}
\justifying
A trainable policy $p_\theta(z \mid x)$ proposes explanation $z$; a frozen evaluator $p^*$ scores it:
\begin{equation}
R(z, x, m;\, p^*) = -\log p^*(z) - \log p^*(x \mid z)
\end{equation}
$p_\theta$ is optimized via REINFORCE/PPO. The frozen $p^*$ prevents reward hacking. \textbf{Planned}: LoRA fine-tuning of Qwen 2.5-7B scored by a larger frozen Qwen instance.
\end{block}

\begin{alertblock}{Proof-of-Concept Scoring (Qwen 2.5-7B)}
\justifying
Sequence: $[1, 1, 2, 5, 12, 27, 58, 121, 248, 503]$

\begin{center}
\begin{tabular}{lcccc}
\toprule
Explanation & Cond.\ $\downarrow$ & Bayes $\downarrow$ & Prior NLL & Lik.\ NLL \\
\midrule
Generated: $a_n = a_{n-1} + 2a_{n-2}$ & 2.37 & 67.66 & 26.87 & 40.79 \\
Concise: $2^n - n$                      & 39.35 & 81.97 & 64.26 & 17.71 \\
Dummy: repeat sequence                 & 63.88 & 106.50 & 106.50 & 0.00 \\
False: $3n + 1$                         & 42.27 & 120.29 & 61.05 & 59.24 \\
\bottomrule
\end{tabular}
\end{center}

\begin{itemize}
  \item Bayesian scoring penalizes the \textbf{dummy} explanation via high prior NLL; conditional loss alone cannot distinguish it
  \item Generated rule achieves the best conditional score: LLM assigns it high probability given the sequence
  \item Bayesian prior correctly exposes the ``repeat sequence'' trivial explanation
\end{itemize}
\end{alertblock}

\begin{block}{Baselines}
\begin{itemize}
  \item \textbf{MDL Policy (Ours)}: RL-optimized $p_\theta(z|x)$. Amortized explanation policy. Full two-part MDL code. No retraining on held-out datasets.
  \item \textbf{Standard ICL}: No explicit explanation. Predictions conditioned on in-context training examples. One-part sequential code.
  \item \textbf{Monte Carlo Search}: Sample / search candidate explanations; score with same MDL objective; no learned policy; per-dataset search.
\end{itemize}
\end{block}

\begin{block}{Related Work}
\begin{itemize}
  \item \textbf{Singh et al.\ (2023)} \cite{singh2023} --- iPrompt: iterative prompt search for input-output mappings. Not MDL-grounded; per-dataset.
  \item \textbf{Ellis (2023)} \cite{ellis2023} --- Bayesian inference over NL hypotheses with LLM prior + likelihood. We extend to an amortized RL policy.
  \item \textbf{Hu et al.\ (2023)} \cite{hu2023} --- Amortized inference for latent text posteriors via RL.
  \item \textbf{Levy et al.\ (2025)} \cite{levy2025} --- WorldLLM (Dataset A); Monte Carlo search baseline.
  \item \textbf{Agarwal et al.\ (2026)} \cite{autodiscovery2025} --- AutoDiscovery (Dataset B).
\end{itemize}
\end{block}

\begin{block}{Conclusions \& Future Work}
\begin{itemize}
  \item We propose an \textbf{MDL framework} for linguistic explanation discovery: a two-part code jointly measures simplicity and predictive power
  \item Proof-of-concept results confirm Bayesian scoring penalizes trivial and false explanations more reliably than conditional scoring alone
  \item \textbf{Amortized RL} training enables the policy to generalize across datasets, avoiding expensive per-dataset search at test time
  \item \textbf{Immediate next step}: LoRA REINFORCE training of Qwen 2.5-7B under the Bayesian reward, with full evaluation on Datasets A and B
  \item \textbf{Future}: multi-modal data, hierarchical explanations, human-in-the-loop evaluation of interpretability
\end{itemize}
\end{block}
