\begin{abstract}
% TODO: Revise after experiments — keep aligned with final results.
Prompted large language models (LLMs) behave like universal function approximators: the prompt acts as natural-language ``parameters'' governing conditional predictions. We frame linguistic explanations as hypotheses about data and apply the Minimum Description Length (MDL) principle using frozen LLM likelihoods. A trainable policy $p_\theta(z \mid x)$ proposes an explanation $z$ for dataset $x$; a frozen evaluator $p^\star$ scores $z$ with a two-part code: explanation complexity $-\log p^\star(z)$ and data complexity $-\log p^\star(x \mid z)$ (with token masking for prediction targets). We optimize $p_\theta$ with reinforcement learning to discover concise, predictive explanations and compare to in-context learning and Monte Carlo hypothesis search. % TODO: Summarize key empirical findings (compression, generalization, interpretability).
\end{abstract}

\subsection{MDL objective and reward}
\label{sec:mdl}

Let $x$ denote a dataset (rendered as text), $z$ a proposed linguistic explanation, and $m$ a mask over tokens that count toward the predictive likelihood (e.g., labels or next-observation fields). With a frozen evaluator LM $p^\star$, we define the reward
\begin{equation}
  R(z, x, m; p^\star) = -\log p^\star(z) - \log p^\star(x \mid z),
  \label{eq:mdl-reward}
\end{equation}
The first term penalizes explanation complexity under $p^\star$; the second penalizes poor prediction of $x$ given $z$, evaluated only on tokens selected by the mask $m$.

\subsection{Baselines}
\label{sec:baselines}

\begin{enumerate}
  \item \textbf{RL-MDL (ours):} policy $p_\theta(z \mid x)$ trained to maximize~\eqref{eq:mdl-reward}.
  \item \textbf{In-context learning (ICL):} no explicit $z$; predictions from training examples in context.
  \item \textbf{Monte Carlo search:} sample or search candidate $z$; select by same MDL score without learned policy.
\end{enumerate}

\subsection{Metrics}
\label{sec:metrics}

\paragraph{Training-set compression.}
For explanation-based methods (ours and search),
\begin{equation}
  -\log p^\star(z) - \sum_{i=1}^{N} \log p^\star(x_i \mid z).
  \label{eq:compression-two-part}
\end{equation}
For ICL, an online one-part code,
\begin{equation}
  -\log p^\star(x_1) - \sum_{i=1}^{N-1} \log p^\star(x_{i+1} \mid x_{1:i}).
  \label{eq:compression-icl}
\end{equation}

\paragraph{Generalization.}
For held-out $x' = (x'_1, \ldots, x'_M)$, explanation-based methods report
\begin{equation}
  \sum_{i=1}^{M} -\log p^\star(x'_i \mid z),
\end{equation}
and ICL reports $\sum_{i=1}^{M} -\log p^\star(x'_i \mid x)$.
